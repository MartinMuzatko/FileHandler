Examples

Batch create tons of folders.

								
$customers = [1,2,3,4,5];
foreach ($customers as $customer)
{
	$file = new File('customer/'.$customer.'/todo.json');
	$file
		->create()
		->write('[{ title: "Read instructions."}]')
		->chmod(0644);
}

Manage uploads without twisting your mind

								
$file = new File($_FILES['file']['tmp_name']);
if ($file->width < 200 && $file->height < 200)
{
	$upload = new File('avatar.'.$file->extension);
	$upload
		->create()
		->write($file)
		->move('customer/1/img');
	echo 'Your avatar has been uploaded';
}
